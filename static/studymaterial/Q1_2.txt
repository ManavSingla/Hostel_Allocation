#include<bits/stdc++.h>
using namespace std;
int main()
{
    long long int n,i,count=0,temp,sum=0,j;
    cin>>n;
    long long int arr[n];
    for (i=0;i<n;i++)
    {
        cin>>arr[i];
    }
    sort(arr,arr+n);
    for(i=1;i<n-1;i++)
    {
        if(arr[i]==arr[i-1])
        {
            arr[i]=arr[i]+1;
            j=i;
            while(arr[j+1]==arr[i-1])
            {
                count++;
                j++;
            }
            {
                temp=arr[i];
                arr[i]=arr[i+count];
                arr[i+count]=temp;
            }
            if(count>0)
            {
                i=i-1;
            }
            count=0;
        }
    }
    if(arr[n-1]==arr[n-2])
    {
        arr[n-1]+=1;
    }
    for(i=0;i<n;i++)
    {
        sum+=arr[i];
    }
    cout<<sum<<endl;
}
